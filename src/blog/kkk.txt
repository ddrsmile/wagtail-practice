from __future__ import absolute_import, unicode_literals

from wagtail.wagtailcore import hooks
from django.contrib.staticfiles.templatetags.staticfiles import static

@hooks.register('insert_editor_js')
def codemirror_mode_js():
    js = [
        static('js/codemirror/lib/codemirror.js'),
        static('js/codemirror/mode/python.js'),
        static('js/codemirror/mode/clike.js'),
        static('js/codemirror/mode/javascript.js'),
        static('js/codemirror/mode/css.js'),
        static('js/codemirror/mode/shell.js'),
    ]

    js_includes = '\n'.join(['\n<script type="text/javascript" src="{filename}"></script>'.format(filename=filename) for filename in js])
    return js_includes

@hooks.register('insert_editor_css')
def codemirror_style_css():
    css = [
        static('css/codemirror/lib/codemirror.css'),
        static('css/codemirror/theme/eclipse.css'),
    ]

    css_includes = '\n'.join(['\n<link type="text/css" rel="stylesheet" href="{filename}">'.format(filename=filename) for filename in css])
    return css_includes
